56   // imperative 
57   def factorialImperative(n: Long): Long = { 
58     var result: Long = 1 
59     for (i: Long <- 1l to n) 
60       result *= i 
61     result 
62   }

65   // normal recursion 
66   def factorialRecursion(n: Long): Long = 
67     if (n == 0) 
68       1 
69     else 
70       n * factorialRecursion(n - 1) 
71   factorialRecursion(4)
